            $("#btn").bind("click", function() {
                var name = $("#txtname").val();
                var pwd = $("#txtpwd").val();
                if (name=== "" || pwd === "")
                { alert("用户名和密码不能为空！"); return false; }
                else {
                    alert(name + "/" + pwd);
                    return true;
                }
            })
        })
